<script>
  // ─────────────────────────────────────────────

  function goTo(n) {
    if (n > current && !validate(current)) return;
    document.getElementById('sec' + current).classList.remove('active');
    current = n;
    document.getElementById('sec' + n).classList.add('active');

  document.getElementById('f-cpf').addEventListener('input', function() {
    let v = this.value.replace(/\D/g,'').slice(0,11);
    if (v.length > 9) v = v.replace(/(\d{3})(\d{3})(\d{3})(\d{1,2})/,'$1.$2.$3-$4');
    else if (v.length > 6) v = v.replace(/(\d{3})(\d{3})(\d+)/,'$1.$2.$3');

    else if (v.length > 2) v = v.replace(/(\d{2})(\d+)/,'$1/$2');

  });

    let v = this.value.replace(/\D/g,'').slice(0,11);
    if (v.length > 7) v = v.replace(/(\d{2})(\d{1})(\d{4})(\d+)/,'$1 $2 $3-$4');

  // UPLOAD DE ARQUIVOS

  const files = { rg: null, selfie: null, cert: null };

  function handleFile(key) {
    const input = document.getElementById('f-' + key);
    const file = input.files[0];
    if (!file) return;

    const maxMB = key === 'cert' ? 10 : 5;
    if (file.size > maxMB * 1024 * 1024) {
      alert('Arquivo muito grande. Tamanho máximo: ' + maxMB + ' MB.');
      input.value = '';
      return;
    }

    const allowed = key === 'cert' ? ['image/jpeg','application/pdf'] : ['image/jpeg'];
    if (!allowed.includes(file.type)) {
      alert(key === 'cert' ? 'Somente PDF ou JPG são aceitos.' : 'Somente arquivos JPG/JPEG são aceitos.');
      input.value = '';
      return;
    }

    files[key] = file;
    const area = document.getElementById('area-' + key);
    area.classList.add('has-file');
    const prev = document.getElementById('prev-' + key);
    prev.classList.add('show');
    document.getElementById('prev-' + key + '-name').textContent = file.name;
    const err = document.getElementById('e-' + key);
    if (err) err.classList.remove('show');
  }

  function removeFile(key) {
    files[key] = null;
    document.getElementById('f-' + key).value = '';
    document.getElementById('area-' + key).classList.remove('has-file');
    document.getElementById('prev-' + key).classList.remove('show');
  }

  function dragOver(e, areaId) {
    e.preventDefault();
    document.getElementById(areaId).classList.add('drag-over');
  }

  function dragLeave(areaId) {
    document.getElementById(areaId).classList.remove('drag-over');
  }

  function dropFile(e, key) {
    e.preventDefault();
    const areaId = 'area-' + key;
    document.getElementById(areaId).classList.remove('drag-over');
    const file = e.dataTransfer.files[0];
    if (!file) return;
    // Simula seleção via input
    const dt = new DataTransfer();
    dt.items.add(file);
    document.getElementById('f-' + key).files = dt.files;
    handleFile(key);
  }

  function validate(sec) {
    if (sec === 1) return true;

    if (sec === 2) {
      let ok = true;
      const nome = document.getElementById('f-nome').value.trim();
      const wpp  = document.getElementById('f-wpp').value.trim();
      const cpf  = document.getElementById('f-cpf').value.trim();
      const nasc = document.getElementById('f-nasc').value.trim();
      const email= document.getElementById('f-email').value.trim();
      const dir  = document.getElementById('f-dir').value;

      showErr('e-nome',  !nome);
      showErr('e-wpp',   wpp.length < 12);
      showErr('e-cpf',   cpf.length < 14);
      showErr('e-nasc',  nasc.length < 10);
      showErr('e-email', !email.includes('@') || !email.includes('.'));
      showErr('e-dir',   !dir);

      if (!nome || wpp.length<12 || cpf.length<14 || nasc.length<10 || !email.includes('@') || !dir) ok = false;
      return ok;

    if (sec === 3) {

      const recert = document.querySelector('input[name="recert"]:checked');
      showErr('e-rg',     !files.rg);
      showErr('e-selfie', !files.selfie);
      showErr('e-recert', !recert);
      if (!files.rg || !files.selfie || !recert) ok = false;
      return ok;
    }

    return true;
  }

  // CONVERSÃO BASE64

  function toBase64(file) {
    return new Promise((res, rej) => {
      const r = new FileReader();
      r.onload = () => res(r.result.split(',')[1]);
      r.onerror = () => rej(new Error('Erro ao ler arquivo'));
      r.readAsDataURL(file);
    });
  }

  // ENVIO

  async function submitForm() {
    if (!validate(3)) return;

    const btn = document.getElementById('btn-submit');
    const overlay = document.getElementById('loading-overlay');
    btn.disabled = true;
    document.getElementById('btn-spin').style.display = 'block';
    document.getElementById('btn-icon').style.display = 'none';
    document.getElementById('btn-text').textContent = 'Enviando…';
    overlay.classList.add('show');

    try {
      const rgBase64     = await toBase64(files.rg);
      const selfieBase64 = await toBase64(files.selfie);
      const certBase64   = files.cert ? await toBase64(files.cert) : null;

      const payload = {
        nome:         document.getElementById('f-nome').value.trim(),
        wpp:          document.getElementById('f-wpp').value.trim(),
        cpf:          document.getElementById('f-cpf').value.trim(),
        nascimento:   document.getElementById('f-nasc').value.trim(),
        email:        document.getElementById('f-email').value.trim(),
        diretoria:    document.getElementById('f-dir').value,
        recertificacao: document.querySelector('input[name="recert"]:checked').value,
        rg: {
          base64: rgBase64,
          name:   files.rg.name,
          type:   files.rg.type
        },
        selfie: {
          base64: selfieBase64,
          name:   files.selfie.name,
          type:   files.selfie.type
        },
        cert: certBase64 ? {
          base64: certBase64,
          name:   files.cert.name,
          type:   files.cert.type
        } : null
      };

      const resp = await fetch(APPS_SCRIPT_URL, {
        method: 'POST',
        body: JSON.stringify(payload)
      });

      const result = await resp.json();

      if (result.status === 'ok') {
        overlay.classList.remove('show');
        goTo(4);
      } else {
        throw new Error(result.message || 'Erro desconhecido');
      }

    } catch (err) {
      overlay.classList.remove('show');
      alert('❌ Erro ao enviar: ' + err.message + '\n\nVerifique sua conexão e tente novamente. Se o problema persistir, entre em contato com May.');
      btn.disabled = false;
      document.getElementById('btn-spin').style.display = 'none';
      document.getElementById('btn-icon').style.display = 'block';
      document.getElementById('btn-text').textContent = 'Enviar inscrição';
    }
  }

  }
